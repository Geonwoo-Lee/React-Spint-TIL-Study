export const Toggler = () => {
  const [theme, setTheme] = useState('light')
  const themeToggler = () => {
  theme === 'light' ? setTheme('dark') : setTheme('light')
}
  return(
    
  <ThemeProvider theme = {theme === 'light' ? lightTheme : darkTheme}>
   <GlobalStyles/>
   <StyledApp>
   <Button onClick = { () => themeToggler()}> Change your mode ! </Button>
  </StyledApp>
  </ThemeProvider>
  )

}
